## 🔍 Verification Commands

### ⚙️ Quick Compatibility Check

```bash
#!/bin/bash
# Yaci Store 2.0 Compatibility Checker

echo "Checking Yaci Store 2.0 Compatibility..."

# Check Java version
JAVA_VERSION=$(java -version 2>&1 | grep -oP '(?<=version ")\\d+' | head -1)
if [ "$JAVA_VERSION" -ge 21 ]; then
    echo "✅ Java $JAVA_VERSION - Compatible"
else
    echo "❌ Java $JAVA_VERSION - Requires Java 21+"
fi

# Check PostgreSQL
PG_VERSION=$(psql --version | grep -oP '\\d+\\.\\d+' | head -1)
if [ "${PG_VERSION%%.*}" -ge 13 ]; then
    echo "✅ PostgreSQL $PG_VERSION - Compatible"
else
    echo "❌ PostgreSQL $PG_VERSION - Requires 13+"
fi

# Check Docker
if command -v docker &> /dev/null; then
    DOCKER_VERSION=$(docker --version | grep -oP '\\d+\\.\\d+' | head -1)
    echo "✅ Docker $DOCKER_VERSION - Available"
else
    echo "🔶 Docker - Not installed (optional)"
fi

# Check system resources
RAM_GB=$(free -g | awk 'NR==2{print $2}')
if [ "$RAM_GB" -ge 4 ]; then
    echo "✅ RAM: ${RAM_GB}GB - Sufficient"
else
    echo "⚠️ RAM: ${RAM_GB}GB - Consider upgrading (4GB+ recommended)"
fi

echo "Compatibility check complete!"
```
